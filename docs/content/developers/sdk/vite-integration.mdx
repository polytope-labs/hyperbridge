# Vite Integration

The Hyperbridge SDK includes a Vite plugin to handle WebAssembly dependencies correctly during development and production builds.

## Overview

The Vite plugin automatically:

- Copies necessary WebAssembly files to the correct location in Vite's dependency cache
- Handles WASM file resolution during development
- Ensures proper bundling for production builds
- Configures logging levels for SDK operations

## Installation

The plugin is included with the SDK package, so no additional installation is required:

```bash
npm install @hyperbridge/sdk
```

## Basic Setup

### vite.config.ts

Add the Hyperbridge plugin to your Vite configuration:

```typescript
import { defineConfig } from "vite"
import react from "@vitejs/plugin-react"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig({
  plugins: [
    react(),
    hyperbridge()
  ]
})
```

### With Options

Configure the plugin with custom options:

```typescript
import { defineConfig } from "vite"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig({
  plugins: [
    hyperbridge({
      logLevel: "trace" // Set SDK log level
    })
  ]
})
```

## Configuration Options

### Plugin Options

| Option | Type | Default | Description |
|--------|------|---------|-------------|
| `logLevel` | `string` | `"info"` | Set the logging level for SDK operations |

### Log Levels

| Level | Description | Use Case |
|-------|-------------|----------|
| `trace` | Most verbose logging | Deep debugging, development |
| `debug` | Detailed debug information | Troubleshooting issues |
| `info` | General information | Default, production |
| `warn` | Warning messages only | Production monitoring |
| `error` | Error messages only | Production, minimal logging |

## Complete Project Setup

### 1. Initialize Project

```bash
# Create Vite project
npm create vite@latest my-hyperbridge-app -- --template react-ts

# Navigate to project
cd my-hyperbridge-app

# Install dependencies
npm install

# Install Hyperbridge SDK
npm install @hyperbridge/sdk
```

### 2. Configure Vite

```typescript
// vite.config.ts
import { defineConfig } from "vite"
import react from "@vitejs/plugin-react"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig({
  plugins: [
    react(),
    hyperbridge({
      logLevel: "info"
    })
  ]
})
```

### 3. Use the SDK

```typescript
// src/App.tsx
import { useEffect, useState } from "react"
import { 
  IndexerClient, 
  createQueryClient, 
  EvmChain,
  SubstrateChain 
} from "@hyperbridge/sdk"

function App() {
  const [indexer, setIndexer] = useState<IndexerClient | null>(null)

  useEffect(() => {
    async function initializeSDK() {
      const queryClient = createQueryClient({
        url: "https://hyperbridge-indexer-url"
      })

      const sourceChain = new EvmChain({
        chainId: 97,
        rpcUrl: "https://data-seed-prebsc-1-s1.binance.org:8545",
        host: "0x...",
        consensusStateId: "BSC0"
      })

      const destChain = new EvmChain({
        chainId: 10200,
        rpcUrl: "https://rpc.chiadochain.net",
        host: "0x...",
        consensusStateId: "GNO0"
      })

      const hyperbridgeChain = new SubstrateChain({
        stateMachineId: "KUSAMA-4009",
        wsUrl: "wss://gargantua.polytope.technology",
        hasher: "Keccak",
        consensusStateId: "PAS0"
      })

      await hyperbridgeChain.connect()

      const client = new IndexerClient({
        queryClient,
        pollInterval: 1_000,
        source: sourceChain,
        dest: destChain,
        hyperbridge: hyperbridgeChain
      })

      setIndexer(client)
    }

    initializeSDK()
  }, [])

  return (
    <div>
      <h1>Hyperbridge App</h1>
      {indexer ? (
        <p>SDK initialized successfully!</p>
      ) : (
        <p>Initializing SDK...</p>
      )}
    </div>
  )
}

export default App
```

## Environment-Specific Configuration

### Development vs Production

```typescript
// vite.config.ts
import { defineConfig } from "vite"
import react from "@vitejs/plugin-react"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig(({ mode }) => ({
  plugins: [
    react(),
    hyperbridge({
      logLevel: mode === "development" ? "debug" : "warn"
    })
  ]
}))
```

### Multiple Environments

```typescript
// vite.config.ts
import { defineConfig, loadEnv } from "vite"
import react from "@vitejs/plugin-react"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig(({ mode }) => {
  const env = loadEnv(mode, process.cwd(), "")
  
  return {
    plugins: [
      react(),
      hyperbridge({
        logLevel: env.VITE_LOG_LEVEL || "info"
      })
    ]
  }
})
```

**.env.development:**

```bash
VITE_LOG_LEVEL=debug
VITE_INDEXER_URL=https://dev-indexer.example.com
```

**.env.production:**

```bash
VITE_LOG_LEVEL=warn
VITE_INDEXER_URL=https://indexer.example.com
```

## Troubleshooting

### WebAssembly Loading Issues

If you encounter WASM loading errors:

```typescript
// vite.config.ts
import { defineConfig } from "vite"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig({
  plugins: [
    hyperbridge({
      logLevel: "trace" // Enable verbose logging
    })
  ],
  optimizeDeps: {
    exclude: ["@hyperbridge/sdk"] // Exclude from pre-bundling if needed
  }
})
```

### Module Resolution Errors

If you see module resolution errors:

```typescript
// vite.config.ts
import { defineConfig } from "vite"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig({
  plugins: [
    hyperbridge()
  ],
  resolve: {
    conditions: ["browser", "module", "import"]
  }
})
```

### Build Errors

For production build issues:

```typescript
// vite.config.ts
import { defineConfig } from "vite"
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig({
  plugins: [
    hyperbridge()
  ],
  build: {
    target: "esnext",
    rollupOptions: {
      output: {
        manualChunks: {
          "hyperbridge-sdk": ["@hyperbridge/sdk"]
        }
      }
    }
  }
})
```

## Common Issues and Solutions

### Issue: WASM file not found

**Solution:** Ensure the Hyperbridge plugin is properly configured:

```typescript
import hyperbridge from "@hyperbridge/sdk/plugins/vite"

export default defineConfig({
  plugins: [
    hyperbridge() // Must be included
  ]
})
```

### Issue: Build size too large

**Solution:** Enable code splitting:

```typescript
export default defineConfig({
  build: {
    rollupOptions: {
      output: {
        manualChunks: {
          "hyperbridge-sdk": ["@hyperbridge/sdk"]
        }
      }
    }
  }
})
```

### Issue: Slow development server

**Solution:** Exclude from optimization:

```typescript
export default defineConfig({
  optimizeDeps: {
    exclude: ["@hyperbridge/sdk"]
  }
})
```
